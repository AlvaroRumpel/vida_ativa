Use a `Column` of rows, each row a `Row` of cells, handling 2 per row with the last odd cell spanning full width.

Algorithm:
```dart
List<Widget> rows = [];
for (int i = 0; i < kpis.length; i += 2) {
  final isLastOdd = i == kpis.length - 1 && kpis.length % 2 == 1;
  if (isLastOdd) {
    rows.add(_buildKpiCell(kpis[i], isFirst: i < 2, isRightCol: false, spanFull: true));
  } else {
    rows.add(IntrinsicHeight(child: Row(crossAxisAlignment: CrossAxisAlignment.stretch, children: [
      Expanded(child: _buildKpiCell(kpis[i], isFirst: i < 2, isRightCol: false)),
      Container(width: 0.5, color: AppTheme.lineHair),  // vertical hairline
      Expanded(child: _buildKpiCell(kpis[i+1], isFirst: i < 2, isRightCol: true)),
    ])));
  }
  if (i + 2 < kpis.length) {
    rows.add(Container(height: 0.5, color: AppTheme.lineHair)); // horizontal hairline
  }
}
return DecoratedBox(
  decoration: BoxDecoration(border: Border.all(color: AppTheme.lineHair, width: 0.5)),
  child: Column(children: rows),
);
```
