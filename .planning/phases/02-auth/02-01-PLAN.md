class AuthLoading extends AuthState {

}

  const AuthUnauthenticated();
  @override
  List<Object?> get props => [];
}

  final String message;
  const AuthError(this.message);

}

**File 2: `lib/features/auth/cubit/auth_cubit.dart`**

Create AuthCubit that:
1. Constructor takes optional `FirebaseAuth` and `FirebaseFirestore` params (for testability). Defaults: `FirebaseAuth.instance`, `FirebaseFirestore.instance`.
2. In constructor body, subscribes to `_auth.authStateChanges().listen(_onAuthStateChanged)`.
3. `_onAuthStateChanged(User? firebaseUser)`:
   - If null: emit `AuthUnauthenticated()`
   - If non-null: fetch `/users/{uid}` doc. If exists: emit `AuthAuthenticated(UserModel.fromFirestore(doc))`. If NOT exists: create the doc as fallback (role='client', displayName from firebaseUser.displayName, email from firebaseUser.email) then emit authenticated. This handles the Pitfall 4 race condition from RESEARCH.md.
4. `signInWithGoogle()`: emit AuthLoading, try `signInWithPopup(GoogleAuthProvider())`, on success create Firestore doc if not exists (same check-then-write pattern), catch `FirebaseAuthException` and emit `AuthError`.
5. `signInWithEmailPassword(String email, String password)`: emit AuthLoading, try `signInWithEmailAndPassword`, catch and emit `AuthError(_mapEmailError(e.code))`.
6. `registerWithEmailPassword({required String name, required String email, required String password})`: emit AuthLoading, try `createUserWithEmailAndPassword`, then `user.updateDisplayName(name)`, then create Firestore doc with UserModel(uid, email, name, role:'client'), catch and emit AuthError.
7. `sendPasswordReset(String email)`: call `_auth.sendPasswordResetEmail(email: email)`. No state change needed â€” caller handles success feedback.
8. `signOut()`: call `_auth.signOut()`. Stream will fire null and emit AuthUnauthenticated.
9. `_mapEmailError(String code)` private method mapping Firebase error codes to Portuguese messages:
   - 'user-not-found' -> 'Email nao encontrado.'
   - 'wrong-password' -> 'Senha incorreta.'
   - 'email-already-in-use' -> 'Email ja cadastrado.'
   - 'weak-password' -> 'Senha muito fraca (minimo 6 caracteres).'
   - 'invalid-email' -> 'Email invalido.'
   - 'invalid-credential' -> 'Email ou senha incorretos.'
   - default -> 'Erro de autenticacao. Tente novamente.'
   (Use proper accented characters: nao->nao, ja->ja, etc. â€” the actual Portuguese: "nao encontrado" -> "nao encontrado", use proper diacritics)
10. Override `close()` to cancel `_authSubscription`.

import 'package:flutter/material.dart';
import 'package:firebase_core/firebase_core.dart';
import 'package:flutter_bloc/flutter_bloc.dart';
import 'package:go_router/go_router.dart';

import 'firebase_options.dart';
import 'core/theme/app_theme.dart';
import 'core/router/app_router.dart';
import 'features/auth/cubit/auth_cubit.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Firebase.initializeApp(options: DefaultFirebaseOptions.currentPlatform);
  runApp(const VidaAtivaApp());
}

class VidaAtivaApp extends StatefulWidget {
  const VidaAtivaApp({super.key});
  @override
  State<VidaAtivaApp> createState() => _VidaAtivaAppState();
}

class _VidaAtivaAppState extends State<VidaAtivaApp> {
  late final AuthCubit _authCubit;
  late final GoRouter _router;

  @override
  void initState() {
    super.initState();
    _authCubit = AuthCubit();
    _router = createRouter(_authCubit);
  }

  @override
  void dispose() {
    _authCubit.close();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return BlocProvider.value(
      value: _authCubit,
      child: MaterialApp.router(
        title: 'Vida Ativa',
        debugShowCheckedModeBanner: false,
        theme: AppTheme.lightTheme,
        routerConfig: _router,
      ),
    );
  }
}
